//@version=5
indicator("Gold AutoResearch Mirror — Next-Day Bias", overlay=false, precision=3)

// ── Inputs ─────────────────────────────────────────────────────────────
train_window   = input.int(750,  "Recent-regime window (bars)", minval=100)
vol_cap        = input.float(3.0,"Vol z-score cap (skip signal)",step=0.1)
up_thresh      = input.float(0.15,"Up threshold (composite)",   step=0.05)
dn_thresh      = input.float(-0.15,"Down threshold (composite)",step=0.05)

w_trend  = input.float(0.30, "Weight: ema_gap / rsi trend")
w_momo   = input.float(0.25, "Weight: price momentum (ret_5d)")
w_dxy    = input.float(0.20, "Weight: DXY momentum (inverse)")
w_vix    = input.float(0.10, "Weight: VIX momentum")
w_real10 = input.float(0.15, "Weight: real-10Y change (inverse)")

// ── External series ────────────────────────────────────────────────────
dxy     = request.security("TVC:DXY",     "D", close, barmerge.gaps_off, barmerge.lookahead_off)
vix     = request.security("CBOE:VIX",    "D", close, barmerge.gaps_off, barmerge.lookahead_off)
real10y = request.security("FRED:DFII10", "D", close, barmerge.gaps_off, barmerge.lookahead_off)

// ── Price features ────────────────────────────────────────────────────
ret_5d   = ta.change(close,5)/ close[5]
vol_20d  = ta.stdev(ta.change(close)/close[1], 20)
ema_10   = ta.ema(close, 10)
ema_50   = ta.ema(close, 50)
ema_gap  = (ema_10 - ema_50) / ema_50
rsi_14   = ta.rsi(close, 14)

// ── Macro momentum ────────────────────────────────────────────────────
dxy_ret_5d     = ta.change(dxy, 5)     / dxy[5]
vix_ret_5d     = ta.change(vix, 5)     / vix[5]
real10y_chg_5d = ta.change(real10y, 5)

// ── Z-normalize over recent-regime window ─────────────────────────────
z(src, len) =>
    m = ta.sma(src, len)
    s = ta.stdev(src, len)
    s == 0 ? 0.0 : (src - m) / s

z_emagap = z(ema_gap,        train_window)
z_rsi    = z(rsi_14 - 50,    train_window)
z_momo   = z(ret_5d,         train_window)
z_dxy    = z(dxy_ret_5d,     train_window)
z_vix    = z(vix_ret_5d,     train_window)
z_real10 = z(real10y_chg_5d, train_window)
z_vol    = z(vol_20d,        train_window)

// ── Composite bias ────────────────────────────────────────────────────
trend_signal = 0.5 * z_emagap + 0.5 * z_rsi
composite    = w_trend  * trend_signal
             + w_momo   * z_momo
             + w_dxy    * (-z_dxy)
             + w_vix    * z_vix
             + w_real10 * (-z_real10)

gated = math.abs(z_vol) > vol_cap
bias  = gated ? 0.0 : composite

long_sig  = bias >  up_thresh and not gated
short_sig = bias <  dn_thresh and not gated

// ── Plots ─────────────────────────────────────────────────────────────
plot(bias, "Composite bias", color=color.new(color.orange, 0), linewidth=2)
hline(up_thresh, "Up threshold", color=color.new(color.green, 40), linestyle=hline.style_dashed)
hline(dn_thresh, "Down threshold", color=color.new(color.red, 40),   linestyle=hline.style_dashed)
hline(0, "Zero", color=color.new(color.gray, 60))

bgcolor(long_sig  ? color.new(color.green, 88) : na)
bgcolor(short_sig ? color.new(color.red,   88) : na)
bgcolor(gated     ? color.new(color.yellow,90) : na)

plotshape(long_sig  and not long_sig[1],  "Flip LONG",  shape.triangleup,   location.bottom, color=color.green, size=size.tiny)
plotshape(short_sig and not short_sig[1], "Flip SHORT", shape.triangledown, location.top,    color=color.red,   size=size.tiny)

// ── Webhook payload (for TradingView alerts → Sentiment Flask) ────────
// In "Create Alert" paste one of these as the Message. Webhook URL:
//   http://<your-host>:5001/api/tv_webhook   (use ngrok to expose)
// LONG:
//   {"secret":"YOUR_SECRET","ticker":"{{ticker}}","signal":"LONG","price":{{close}},"bias":{{plot_0}},"time":"{{timenow}}"}
// SHORT:
//   {"secret":"YOUR_SECRET","ticker":"{{ticker}}","signal":"SHORT","price":{{close}},"bias":{{plot_0}},"time":"{{timenow}}"}

alertcondition(long_sig  and not long_sig[1],  "Gold next-day LONG",  "Composite bias flipped long.")
alertcondition(short_sig and not short_sig[1], "Gold next-day SHORT", "Composite bias flipped short.")
